libutils.formatUrl = (urlObj) ->
  query = urlObj.query || {}
  sep = '&'
  eq = '='
  pairs = _.map query, (value, key) ->
    return encodeURIComponent(key) + eq + encodeURIComponent(value)
  qs = pairs.join sep
  url = urlObj.path || '/'
